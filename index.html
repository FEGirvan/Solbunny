app.post("/feed", async (req, res) => { try { const wallets = req.body.wallets?.slice(0,5) || [] let allTx = [] for (const w of wallets) { const url = `https://api.helius.xyz/v0/addresses/${w}/transactions?api-key=${HELIUS_KEY}` const data = await fetch(url).then(r=>r.json()) data.slice(0,15).forEach(tx=>{ const amt = tx.nativeTransfers?.[0]?.amount ? tx.nativeTransfers[0].amount / 1e9 : 0 if(amt < 2) return // filter noise allTx.push({ wallet: w, amt, time: tx.timestamp }) }) } let signals = [] // ===================== // 🧠 SCORING ENGINE // ===================== let score = 0 // 1. SIZE SCORE allTx.forEach(t=>{ if(t.amt > 20) score += 15 else if(t.amt > 10) score += 10 else if(t.amt > 5) score += 5 }) // 2. FREQUENCY SCORE const now = Math.floor(Date.now()/1000) const recent = allTx.filter(t=> now - t.time < 60) if(recent.length >= 6) score += 20 else if(recent.length >= 3) score += 10 // 3. WALLET DIVERSITY const uniqueWallets = new Set(allTx.map(t=>t.wallet)) if(uniqueWallets.size >= 4) score += 20 else if(uniqueWallets.size >= 2) score += 10 // 4. CONSISTENCY const grouped = {} allTx.forEach(t=>{ if(!grouped[t.wallet]) grouped[t.wallet] = [] grouped[t.wallet].push(t.amt) }) Object.values(grouped).forEach(txs=>{ if(txs.length >= 3) score += 10 }) // LIMIT SCORE if(score > 100) score = 100 // ===================== // 🎯 SIGNAL OUTPUT // ===================== if(score >= 70){ signals.push({ message:`🚀 STRONG SIGNAL (${score})`, level:"danger" }) } else if(score >= 40){ signals.push({ message:`🔥 MEDIUM SIGNAL (${score})`, level:"pump" }) } else if(score >= 20){ signals.push({ message:`👀 WEAK SIGNAL (${score})`, level:"whale" }) } // ===================== // DETAIL SIGNAL // ===================== allTx.forEach(t=>{ if(t.amt > 25){ signals.push({ message:`🐋 Big Buy ${t.amt.toFixed(2)} SOL`, level:"danger" }) } else if(t.amt > 10){ signals.push({ message:`🚨 Strong Buy ${t.amt.toFixed(2)} SOL`, level:"pump" }) } else if(t.amt > 5){ signals.push({ message:`👀 Accumulation ${t.amt.toFixed(2)} SOL`, level:"whale" }) } }) res.json(signals) } catch { res.json([]) } })
